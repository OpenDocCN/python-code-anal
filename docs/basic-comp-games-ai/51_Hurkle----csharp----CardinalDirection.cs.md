# 定义名为hurkle的命名空间，用于组织和管理相关的类和枚举
namespace hurkle
{
    # 定义名为CardinalDirection的枚举类型，用于表示基本的方向
    internal enum CardinalDirection
    {
        # 表示没有方向
        None,
        # 表示北方向
        North,
        # 表示东北方向
        NorthEast,
        # 表示东方向
        East,
        # 表示东南方向
        SouthEast,
        # 表示南方向
        South,
        # 表示西南方向
        SouthWest,
        # 表示西方向
        West,
        # 表示西北方向
        NorthWest
    }
}
